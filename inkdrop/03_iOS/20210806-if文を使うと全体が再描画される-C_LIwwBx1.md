# if文を使うと全体が再描画される

Group内でif文で画面を切り替えると、全体が再描画される。

```swift

class ViewModel: ObservableObject {
    @Published var count = 0    
    func increment() {
        count += 1
    }    
}

struct ContentView: View {
    
    @ObservedObject var viewModel = ViewModel()
    
    var body: some View {
        
        Group {
            
            if viewModel.count == 0 {
                HStack {
                    Text("[Binding]Increment")
                    Button(action: { viewModel.increment() }, label: {
                        Image(systemName: "plus")
                    })
                    Text("\(viewModel.count)")
                }
            } else if viewModel.count == 1 {
                HStack {
                    Text("[Binding]Increment")
                    Button(action: { viewModel.increment() }, label: {
                        Image(systemName: "plus")
                    })
                    Text("\(viewModel.count)")
                }
            } else {
                HStack {
                    Text("[Binding]Increment")
                    Button(action: { viewModel.increment() }, label: {
                        Image(systemName: "plus")
                    })
                    Text("\(viewModel.count)")
                }
            }
            
        }.onAppear {
            debugPrint("*** onAppear")
            //if文でonAppearが何回も呼ばれる。全体が再描画
        }
        
    }
}
```
